Método Filter

  // metodo utilizado para criar um novo array apos filtrar todos os elementos elementos de um array, que respeite as condicoes
  
  // array inicial para análise 
  const idades = [23, 18, 12, 37, 53, 70, 16]
  
  
  // 1º modo
  const maior = idades.filter((valor)=>{
    if(valor >= 18){
      return valor
    }
  })
  console.log(maior)
  
  
  // 2º modo 
  const menorIdade = (valor) => {
    if (valor < 18) {
      return valor
    }
  }
  const menor = idades.filter(menorIdade)
  console.log(menor)
  
  
  // 3º modo
  function ficouMaior(valor){
    if(valor === 18){
      return valor
    }
  }
  const novato = idades.filter(ficouMaior)
  console.log(novato)
