Funções Geradoras

    function* cores(){
    yield 'Verde'
    yield 'Amarelo'
    yield 'Azul'
  }
  
  const itc = cores()
  console.log(itc.next().value)
  console.log(itc.next().value)
  console.log(itc.next().value)
  console.log(itc.next().value)
  
  function* perguntas(){
    const nome = yield 'Qual seu nome?'
    const idade = yield 'Qual sua idade?'
    return 'Seu nome é ' + nome + ', sua idade é ' + idade
  }
  const itr = perguntas()
  console.log(itr.next().value)
  console.log(itr.next('Fulano').value)
  console.log(itr.next(99).value)
  
  
  function* contador(){
    let i = 0
    
    while(true){
      yield i++
      if(i > 8){
        break
      }
    }
  }
  const ite = contador()
  console.log(ite.next().value)
  console.log(ite.next().value)
  console.log(ite.next().value)
  for(let i = 0; i < 10; i++){
    console.log(ite.next().value)
  }
  
  for(let c of ite){
    console.log(c)
  }
